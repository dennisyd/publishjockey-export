% Simple header file for formatting adjustments
% Bare minimum to avoid LaTeX errors

% Set paragraph spacing
\setlength{\parindent}{0pt}    % No paragraph indentation
\setlength{\parskip}{1em}      % Space between paragraphs

% Simple section styling that won't conflict
\makeatletter
\renewcommand{\section}{\@startsection{section}{1}{0pt}%
  {-2ex plus -1ex minus -.2ex}%
  {0.5ex plus .2ex}%
  {\normalfont\Large\bfseries\centering}}
\makeatother

% Adjust chapter heading spacing in a way that doesn't conflict
\makeatletter
\def\@makechapterhead#1{%
  \vspace*{10\p@}%
  {\parindent \z@ \centering \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \if@mainmatter
        \huge\bfseries \@chapapp\space \thechapter
        \par\nobreak
        \vskip 5\p@
      \fi
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak
    \vskip 10\p@
  }}
\makeatother

% Custom chapter spacing
\AtBeginDocument{
  % Adjust spacing for chapters (will only apply inside the document)
  \titlespacing*{\chapter}
    {0pt}       % left
    {0pt}       % before - minimal space before chapter
    {10pt}      % after - reduced space after chapter title
    
  % Adjust spacing for sections
  \titlespacing*{\section}
    {0pt}       % left
    {5pt}       % before - minimal space before section
    {5pt}       % after - minimal space after section
}

% Ensure headings are properly centered and formatted
\renewcommand{\chaptitlefont}{\normalfont\huge\bfseries\centering}
\renewcommand{\chapnumfont}{\normalfont\huge\bfseries\centering}
\renewcommand{\secheadstyle}{\normalfont\Large\bfseries\centering}

% Redefine chapter and section spacing for better display
\usepackage{titlesec}

\titleformat{\chapter}[display]
  {\normalfont\huge\bfseries\centering}
  {\chaptertitlename\ \thechapter}
  {10pt}
  {\Huge}
  [\vspace{5pt}]

% Section format (level below chapter) - tighter spacing
\titleformat{\section}
  {0pt}        % left
  {5pt}        % before - minimal space before section heading
  {5pt}        % after - minimal space after section heading

% Force geometry package to use the correct paper size
\geometry{
  paperwidth=\paperwidth,
  paperheight=\paperheight,
  margin=1in,
  bindingoffset=0mm

% Adjust chapter heading spacing in a way that doesn't conflict
\makeatletter
\def\@makechapterhead#1{%
  \vspace*{10\p@}%
  {\parindent \z@ \centering \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \if@mainmatter
        \huge\bfseries \@chapapp\space \thechapter
        \par\nobreak
        \vskip 5\p@
      \fi
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak
    \vskip 10\p@
  }}
\makeatother

% Custom chapter spacing
\AtBeginDocument{
  % Adjust spacing for chapters (will only apply inside the document)
  \titlespacing*{\chapter}
    {0pt}       % left
    {0pt}       % before - minimal space before chapter
    {10pt}      % after - reduced space after chapter title
    
  % Adjust spacing for sections
  \titlespacing*{\section}
    {0pt}       % left
    {5pt}       % before - minimal space before section
    {5pt}       % after - minimal space after section
}

% Redefine chapter and section spacing for better display
\usepackage{titlesec}

% Chapter format - make it more compact
\titleformat{\chapter}[display]
  {\normalfont\huge\bfseries\centering}
  {\chaptertitlename\ \thechapter}
  {10pt}
  {\Huge}
  [\vspace{5pt}]

% Section format (level below chapter) - tighter spacing
\titleformat{\section}
  {0pt}        % left
  {5pt}        % before - minimal space before section heading
  {5pt}        % after - minimal space after section heading

% Force geometry package to use the correct paper size
\geometry{
  paperwidth=\paperwidth,
  paperheight=\paperheight,
  margin=1in,
  bindingoffset=0mm
}

% Fix for header spacing
\pagestyle{plain}
% \pagenumbering{arabic}  % <-- Commented out to allow \frontmatter/\mainmatter to control numbering

% Adjust chapter heading spacing in a way that doesn't conflict
\makeatletter
\def\@makechapterhead#1{%
  \vspace*{10\p@}%
  {\parindent \z@ \centering \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \if@mainmatter
        \huge\bfseries \@chapapp\space \thechapter
        \par\nobreak
        \vskip 5\p@
      \fi
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak
    \vskip 10\p@
  }}
\makeatother

% Custom chapter spacing
\AtBeginDocument{
  % Adjust spacing for chapters (will only apply inside the document)
  \titlespacing*{\chapter}
    {0pt}       % left
    {0pt}       % before - minimal space before chapter
    {10pt}      % after - reduced space after chapter title
    
  % Adjust spacing for sections
  \titlespacing*{\section}
    {0pt}       % left
    {5pt}       % before - minimal space before section
    {5pt}       % after - minimal space after section
}

% Adjust chapter heading spacing in a way that doesn't conflict
\makeatletter
\def\@makechapterhead#1{%
  \vspace*{10\p@}%
  {\parindent \z@ \centering \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \if@mainmatter
        \huge\bfseries \@chapapp\space \thechapter
        \par\nobreak
        \vskip 5\p@
      \fi
    \fi
    \interlinepenalty\@M
    \Huge \bfseries #1\par\nobreak
    \vskip 10\p@
  }}
\makeatother

  % Adjust spacing for chapters (will only apply inside the document)
  \titlespacing*{\chapter}
    {0pt}       % left
    {0pt}       % before - minimal space before chapter
    {10pt}      % after - reduced space after chapter title
    
  % Adjust spacing for sections
  \titlespacing*{\section}
    {0pt}       % left
    {5pt}       % before - minimal space before section
    {5pt}       % after - minimal space after section
}
